export default {
  async fetch(req: Request, env: Env, ctx: ExecutionContext) {
    if (req.method === "OPTIONS") return okCors(env);
    if (new URL(req.url).pathname === "/api/health") return json({ ok: true });
    if (new URL(req.url).pathname === "/api/generate" && req.method === "POST") {
      const origin = req.headers.get("origin") || "";
      if (!env.ALLOWED_ORIGINS.split(",").includes(origin)) return new Response("forbidden", { status: 403 });
      // rate limit check (KV or Ratelimit API) — pseudo
      // const allowed = await checkRate(req, env);
      // if (!allowed) return new Response("rate", { status: 429, headers: { "Retry-After": "60" } });
      const body = await req.json();
      const { ok, err } = validate(body);
      if (!ok) return json({ error: err }, 400);
      const prompt = buildPrompt(body);
      const ai = await generate(prompt, env.OPENAI_API_KEY); // call model
      const md = postProcess(ai, body.options);
      return cors(json({ markdown: md, labels: suggest(md), usage: { countToday: 3, limit: 10 } }), origin);
    }
    return new Response("not found", { status: 404 });
  }
} satisfies ExportedHandler;

function json(data: any, status = 200) {
  return new Response(JSON.stringify(data), { status, headers: { "content-type": "application/json; charset=utf-8" } });
}
function cors(res: Response, origin = "*") {
  const h = new Headers(res.headers);
  h.set("Access-Control-Allow-Origin", origin);
  h.set("Access-Control-Allow-Headers", "content-type, authorization");
  h.set("Access-Control-Allow-Methods", "POST, OPTIONS, GET");
  return new Response(res.body, { status: res.status, headers: h });
}
function okCors(env: any) {
  return new Response(null, {
    status: 204,
    headers: {
      "Access-Control-Allow-Origin": "*",
      "Access-Control-Allow-Headers": "content-type, authorization",
      "Access-Control-Allow-Methods": "POST, OPTIONS, GET",
      "Access-Control-Max-Age": "600"
    }
  });
}
